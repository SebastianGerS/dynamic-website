CREATE TABLE users (
    id INT UNSIGNED NOT NULL PRIMARY KEY AUTO_INCREMENT,
    firstname VARCHAR(20) NOT NULL,
    surname VARCHAR(20) NOT NULL,
    email VARCHAR(20) NOT NULL UNIQUE,
    username VARCHAR(20) NOT NULL UNIQUE,
    password VARCHAR(20) NOT NULL,
    type ENUM ("admin", "user")
    );

CREATE TABLE blogposts_info (
    id INT UNSIGNED NOT NULL PRIMARY KEY AUTO_INCREMENT,
    user_id INT UNSIGNED NOT NULL,
    post_name VARCHAR(40) NOT NULL,
    post_creation_time DATETIME NOT NULL,
    post_edit_time DATETIME,
    FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE blogposts_content (
    id INT UNSIGNED NOT NULL,
    content VARCHAR(8000) NOT NULL,
    FOREIGN KEY (id) REFERENCES blogposts_info(id)
);

CREATE TABLE blogposts_comments (
    id INT UNSIGNED NOT NULL PRIMARY KEY AUTO_INCREMENT,
    post_id INT UNSIGNED NOT NULL,
    content VARCHAR(4000) NOT NULL,
    user_id INT UNSIGNED NOT NULL,
    post_creation_time DATETIME NOT NULL,
    post_edit_time DATETIME,
    FOREIGN KEY (post_id) REFERENCES blogposts_info(id),
    FOREIGN KEY (user_id) REFERENCES users(id)
);

CREATE TABLE tags(
    id INT UNSIGNED AUTO_INCREMENT,
    tagname VARCHAR(20) NOT NULL
);

CREATE TABLE post_tag_correspondens (
    post_id INT UNSIGNED,
    tag_id INT UNSIGNED,
    FOREIGN KEY (post_id) REFERENCES blogposts_info(id),
    FOREIGN KEY (tag_id) REFERENCES tags(id)
);
